/******************************  Operators in Javascript Example   **************************************
Arithmatic Operator

  (+) operation

  var num1 = 15;
  var num2 = 5;
  var num3 = num1 + num2;
  console.log(num3);  // 20

   (-) operation
  
  var num1 = 15;
  var num2 = 5;
  var num3 = num1 - num2;
  console.log(num3);  // 10

   (*) operation
  
  var num1 = 5;
  var num2 = 5;
  var num3 = num1 * num2;
  console.log(num3);  // 25

   (/) operation --> gives remainder
  
  var num1 = 15;
  var num2 = 5;
  var num3 = num1 / num2;
  console.log(num3);  // 3

   (%) operation  --> gives quotient
  
  var num1 = 15;
  var num2 = 5;
  var num3 = num1 % num2;
  console.log(num3);  // 0

   (++) operation  --> post increment
  
  var num1 = 15;
  var a = num1++;
  console.log(a);  // 16

   (++) operation --> pre increment
  
  var num1 = 15;
  var a = ++num1;
  console.log(a);  // 16
